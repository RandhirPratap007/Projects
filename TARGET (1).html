# Import libraries
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Sample DataFrame Initialization (Replace this with your actual data loading)
# Example: data = pd.read_csv('your_file.csv')
data = pd.DataFrame({
    'category': ['A', 'A', 'B', 'B', 'C', 'C'],
    'price': [100, 150, 200, 250, 300, 350]
})

# Verify DataFrame Columns
print("Data Columns:", data.columns)
print(data.head())  # Check if data is loaded correctly

# Visualization
plt.figure(figsize=(10, 6))
for category in data['category'].unique():
    sns.histplot(data[data['category'] == category]['price'], 
                 kde=True, 
                 label=category, 
                 alpha=0.6)
    
plt.title("Histogram: Price Distribution by Category")
plt.xlabel("Price")
plt.ylabel("Frequency")
plt.legend(title="Category")
plt.show()
